public class SaveTrigger : MonoBehaviour
{
    public void LoadLevel()
    {
        if (PlayerPrefs.HasKey("level"))
        {
            SceneManager.LoadScene(PlayerPrefs.GetInt("level", +1));
        }
        else
        {
            SceneManager.LoadScene(1);
        }
    }
}
